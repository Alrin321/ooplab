import java.util.*;
class Armstrong
{
	public static void main(String args[])
	{
		int num,sum=0,rem,n;
		
		Scanner sc =new Scanner(System.in);
		System.out.println("Enter the Number:");
		num = sc.nextInt();
		n=num;
		while(n!=0)
		{
			rem=n%10;
			sum+=rem*rem*rem;
			n/=10;
		}

		 if(sum==num)
		 {
		 	System.out.println("the number is armstrong");
		 }
		 else
		 {
		 	System.out.println("the number is not armstrong");
		 }

		 }


        }
